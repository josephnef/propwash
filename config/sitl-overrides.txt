# SITL-specific neutralisation applied AFTER a real FC diff, for settings
# that describe the physical board and cannot apply to the virtual sim.
#
#  - Board alignment: the real FC is mounted inverted (align_board_roll = 180
#    in the diff). That rotation compensates for how the gyro chip sits in
#    the airframe. propwash's virtual gyro already feeds airframe-frame data,
#    so leaving align_board_roll = 180 would rotate correct data by 180° and
#    the quad would fly inverted / refuse to self-level. Zero it.
set align_board_roll = 0
set align_board_yaw = 0
